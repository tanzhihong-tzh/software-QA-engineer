云链电商平台UI自动化回归测试

1. 概述

一个基于Selenium和Python的UI自动化回归测试框架，用于测试云链电商平台的核心功能。项目采用了Page Object Model (POM) 设计模式，将页面元素定位、操作和业务逻辑分离，提高了测试用例的可维护性和可扩展性。

目标：通过自动化测试确保云链电商平台的核心功能在持续开发和迭代过程中保持稳定，减少人工回归测试的工作量，提高测试效率和质量。

2. 技术架构

2.1 核心技术栈

  • 自动化测试框架：Selenium WebDriver
  • 编程语言：Python 3.9
  • 测试运行器：pytest
  • 设计模式：Page Object Model (POM)

2.2 项目架构

项目采用了分层架构设计，主要分为以下几层：

  1. 工具层 (util)：提供通用的工具函数和浏览器驱动管理
  2. 页面对象层 (Pages)：封装页面元素定位和基础操作
  3. 业务逻辑层：封装业务流程和测试场景
  4. 测试用例层 (Shop_UI)：实现具体的测试用例

4.1 工具层 (util.py)

工具层提供了通用的工具函数和浏览器驱动管理，主要包括：

  • Driver_util类：实现浏览器驱动的单例模式管理，确保整个测试过程中只使用一个浏览器实例
  • get_alter_msg函数：获取弹窗消息

util.py - 完整代码

from time import sleep
from selenium import webdriver
from selenium.webdriver.common.by import By


def get_alter_msg():
    sleep(2)
    msg = Driver_util.get_chrome().find_element(By.CLASS_NAME, 'layui-layer-content').text
    return msg

class Driver_util(object):

    __driver = None

    @classmethod
    def get_chrome(cls):
        if cls.__driver is None:
            cls.__driver = webdriver.Chrome()
            cls.__driver.get("http://192.168.11.131/index.php")
            cls.__driver.maximize_window()
            cls.__driver.implicitly_wait(10)
        return cls.__driver

    @classmethod
    def close_driver(cls):
        if cls.__driver is not None:
            sleep(2)
            cls.__driver.quit()
            cls.__driver = None

if __name__ == '__main__':
    Driver_util.get_chrome()
    sleep(2)
    Driver_util.close_driver()

"""
登录页面

from selenium.webdriver.common.by import By

from util import Driver_util


class Login_Page(object):#创建定位元素对象类
    def __init__(self):#获取浏览器对象
        # 说明：将元素的定位方法及特征值封装成属性，自能够实现集中中管理目标元素的定位方法及特征值
        self.driver = Driver_util.get_chrome()
        self.name = (By.ID, 'username')#用户名
        self.pwd = (By.ID, 'password')#密码
        self.code = (By.ID, 'verify_code')#验证码
        self.login = (By.NAME, 'sbtbutton')#登录按钮
        self.yes_01 =(By.CLASS_NAME,'layui-layer-btn0')#确认按钮
        self.index_01 = (By.XPATH,'/html/body/div[1]/div/div[1]/a')#首页按钮
    def find_username(self):#定位用户名元素
        return self.driver.find_element(self.name[0],self.name[1])
    def find_password(self):#定位密码元素
        return self.driver.find_element(self.pwd[0],self.pwd[1])
    def find_verify_code(self):#定位验证码元素
        return self.driver.find_element(self.code[0],self.code[1])
    def click_login(self):#定位登录按钮元素
        return self.driver.find_element(self.login[0],self.login[1])
    def yes(self):
        return self.driver.find_element(self.yes_01[0],self.yes_01[1])
    def find_index(self):
        return self.driver.find_element(self.index_01[0],self.index_01[1])

class Login_handler(object):#创建操作对象类
    def __init__(self):#获取定位元素对象

        self.find.click_login().click()
    def click_yes(self):
        self.find.yes().click()
    def click_index(self):
        self.find.find_index().click()

4.2.2 购物车页面对象 (cart_page.py)

封装购物车页面的元素定位和操作，支持添加商品、修改数量、删除商品、清空购物车等功能。

cart_page.py - 完整代码

"""
购物车页面
"""
from selenium.webdriver.common.by import By
from util import Driver_util


class Cart_Page(object):#创建定位元素对象类
    def __init__(self):#获取浏览器对象
        self.driver = Driver_util.get_chrome()
        # 购物车商品相关元素
        self.cart_items = (By.CSS_SELECTOR, '.u-fn-cart .c-item ')#购物车商品项
        self.checkbox_all = (By.CLASS_NAME, 'checkall checkFull checkall-true ')#全选复选框
        self.checkbox_item = (By.CSS_SELECTOR, '.checkall')#单个商品复选框
        # 商品信息
        self.product_names = (By.CSS_SELECTOR, '.detail-ggsl')#商品名称
        self.product_prices = (By.CSS_SELECTOR, '.shop-price-cou .shop-price-le ul .bigpri_jj')#商品单价
        self.product_nums = (By.CSS_SELECTOR, '.standard .minus-plus input ')#商品数量
        # 操作按钮
        self.reduce_btn = (By.CLASS_NAME, 'mins')#减少数量按钮
        self.add_btn = (By.CLASS_NAME, 'add')#增加数量按钮
        self.delete_btn = (By.CLASS_NAME, 'deleteGoods deleteItem')#删除商品按钮
        self.clear_cart_btn = (By.ID, 'clearCart')#清空购物车按钮
        # 结算相关
        self.total_price = (By.ID, 'cart_2_total_price')#合计价格
        self.checkout_btn = (By.CLASS_NAME, 'paytotal')#结算按钮
        # 空购物车提示
        self.empty_cart_tip = (By.CLASS_NAME, 'txt nologin')#空购物车提示
    
    def find_checkbox_all(self):#定位全选复选框
        return self.driver.find_element(self.checkbox_all[0], self.checkbox_all[1])
    
    def find_checkbox_item(self):#定位所有单个商品复选框
        return self.driver.find_elements(self.checkbox_item[0], self.checkbox_item[1])
    
    def find_cart_items(self):#定位所有购物车商品项
        return self.driver.find_elements(self.cart_items[0], self.cart_items[1])
    
    def find_product_names(self):#定位所有商品名称
        return self.driver.find_elements(self.product_names[0], self.product_names[1])
    
    def find_product_prices(self):#定位所有商品单价
        return self.driver.find_elements(self.product_prices[0], self.product_prices[1])
    
    def find_product_nums(self):#定位所有商品数量输入框
        return self.driver.find_elements(self.product_nums[0], self.product_nums[1])
    
    def find_reduce_btn(self):#定位所有减少数量按钮
        return self.driver.find_elements(self.reduce_btn[0], self.reduce_btn[1])
    
    def find_add_btn(self):#定位所有增加数量按钮
        return self.driver.find_elements(self.add_btn[0], self.add_btn[1])
    
    def find_delete_btn(self):#定位所有删除商品按钮
        return self.driver.find_elements(self.delete_btn[0], self.delete_btn[1])
    
    def find_clear_cart_btn(self):#定位清空购物车按钮
        return self.driver.find_element(self.clear_cart_btn[0], self.clear_cart_btn[1])
    
    def find_total_price(self):#定位合计价格
        return self.driver.find_element(self.total_price[0], self.total_price[1])
    
    def find_checkout_btn(self):#定位结算按钮
        return self.driver.find_element(self.checkout_btn[0], self.checkout_btn[1])
    
    def find_empty_cart_tip(self):#定位空购物车提示
        return self.driver.find_element(self.empty_cart_tip[0], self.empty_cart_tip[1])


class Cart_handler(object):#创建操作对象类
    def __init__(self):#获取定位元素对象
        self.find = Cart_Page()
    
    def click_checkbox_all(self):#点击全选复选框
        self.find.find_checkbox_all().click()
    
    def click_checkbox_item(self, index):#点击指定索引的商品复选框
        items = self.find.find_checkbox_item()
        if index < len(items):
            items[index].click()
    
    def get_cart_items_count(self):#获取购物车商品数量
        return len(self.find.find_cart_items())
    
    def get_product_names(self):#获取所有商品名称
        return [item.text for item in self.find.find_product_names()]
    
    def get_product_prices(self):#获取所有商品单价
        return [item.text for item in self.find.find_product_prices()]
    
    def get_product_nums(self):#获取所有商品数量
        return [item.get_attribute('value') for item in self.find.find_product_nums()]
    
    def set_product_num(self, index, num):#设置指定商品的数量
        nums = self.find.find_product_nums()
        if index < len(nums):
            nums[index].clear()
            nums[index].send_keys(num)
    
    def click_reduce_btn(self, index):#点击指定商品的减少数量按钮
        btns = self.find.find_reduce_btn()
        if index < len(btns):
            btns[index].click()
    
    def click_add_btn(self, index):#点击指定商品的增加数量按钮
        btns = self.find.find_add_btn()
        if index < len(btns):
            btns[index].click()
    
    def click_delete_btn(self, index):#点击指定商品的删除按钮
        btns = self.find.find_delete_btn()
        if index < len(btns):
            btns[index].click()
    
    def click_clear_cart(self):#点击清空购物车按钮
        self.find.find_clear_cart_btn().click()
    
    def get_total_price(self):#获取合计价格
        return self.find.find_total_price().text
    
    def click_checkout(self):#点击结算按钮
        self.find.find_checkout_btn().click()
    
    def is_cart_empty(self):#判断购物车是否为空
        try:
            return self.find.find_empty_cart_tip().is_displayed()
        except:
            return False


class Cart_task(object):#创建业务对象类
    def __init__(self):#获取操作对象
        self.handler = Cart_handler()
        self.driver = self.handler.find.driver
    
    def task_go_to_cart(self):#进入购物车页面业务
        self.driver.get("http://192.168.11.131/Home/Cart/index.html")
    
    def task_select_all_products(self):#全选商品业务
        self.handler.click_checkbox_all()
    
    def task_select_product(self, index):#选择指定商品业务
        self.handler.click_checkbox_item(index)
    
    def task_update_product_num(self, index, num):#更新商品数量业务
        self.handler.set_product_num(index, num)
    
    def task_delete_product(self, index):#删除商品业务
        self.handler.click_delete_btn(index)
    
    def task_clear_cart(self):#清空购物车业务
        self.handler.click_clear_cart()
    
    def task_checkout(self):#结算业务
        self.handler.click_checkout()
    
    def task_get_cart_info(self):#获取购物车信息业务
        return {
            'count': self.handler.get_cart_items_count(),
            'names': self.handler.get_product_names(),
            'prices': self.handler.get_product_prices(),
            'nums': self.handler.get_product_nums(),
            'total_price': self.handler.get_total_price(),
            'is_empty': self.handler.is_cart_empty()
        }

4.2.3 搜索页面对象 (search_page.py)

封装搜索页面的元素定位和操作，支持关键词搜索、查看搜索结果、查看商品详情等功能。

search_page.py - 完整代码

"""
商品搜索页面
"""
from selenium.webdriver.common.by import By
from util import Driver_util


class Search_Page(object):#创建定位元素对象类
    def __init__(self):#获取浏览器对象
        self.driver = Driver_util.get_chrome()
        # 搜索相关元素
        self.search_input = (By.ID, 'q')#搜索输入框
        self.search_btn = (By.CLASS_NAME, 'ecsc-search-button')#搜索按钮
        # 搜索结果相关元素
        self.search_results = (By.CSS_SELECTOR, '.goods-list')#搜索结果列表
        self.product_items = (By.CSS_SELECTOR, '.xs_img')#商品项
        self.product_names = (By.CSS_SELECTOR, 'old shop_name2')#商品名称（修改为直接定位到a标签）
        self.product_prices = (By.CSS_SELECTOR, '.li_xfo')#商品价格
        # 分页相关元素
        self.pagination = (By.CSS_SELECTOR, '.pagination')#分页控件
        self.next_page = (By.ID, 'example1_next')#下一页按钮
    
    # 搜索功能元素定位
    def find_search_input(self):#定位搜索输入框
        return self.driver.find_element(self.search_input[0], self.search_input[1])
    
    def find_search_btn(self):#定位搜索按钮
        return self.driver.find_element(self.search_btn[0], self.search_btn[1])
    
    # 搜索结果元素定位
    def find_search_results(self):#定位搜索结果列表
        return self.driver.find_element(self.search_results[0], self.search_results[1])
    
    def find_product_items(self):#定位所有商品项
        return self.driver.find_elements(self.product_items[0], self.product_items[1])
    
    def find_product_names(self):#定位所有商品名称
        return self.driver.find_elements(self.product_names[0], self.product_names[1])
    
    def find_product_prices(self):#定位所有商品价格
        return self.driver.find_elements(self.product_prices[0], self.product_prices[1])
    
    # 分页元素定位
    def find_pagination(self):#定位分页控件
        return self.driver.find_element(self.pagination[0], self.pagination[1])
    
    def find_next_page(self):#定位下一页按钮
        return self.driver.find_element(self.next_page[0], self.next_page[1])


class Search_handler(object):#创建操作对象类
    def __init__(self):#获取定位元素对象
        self.find = Search_Page()
    
    def send_search_keyword(self, keyword):#输入搜索关键词
        self.find.find_search_input().clear()
        self.find.find_search_input().send_keys(keyword)
    
    def click_search_btn(self):#点击搜索按钮
        self.find.find_search_btn().click()
    
    def get_product_items_count(self):#获取商品数量
        return len(self.find.find_product_items())
    
    def get_product_names(self):#获取所有商品名称
        return [item.text for item in self.find.find_product_names()]
    
    def get_product_prices(self):#获取所有商品价格
        return [item.text for item in self.find.find_product_prices()]
    
    def click_product_by_index(self, index):#点击指定索引的商品
        items = self.find.find_product_items()
        if index < len(items):
            items[index].click()
    
    def click_next_page(self):#点击下一页
        self.find.find_next_page().click()


class Search_task(object):#创建业务对象类
    def __init__(self):#获取操作对象
        self.handler = Search_handler()
        self.driver = self.handler.find.driver
    
    def task_search(self, keyword):#搜索业务层
        self.handler.send_search_keyword(keyword)#输入搜索关键词业务
        self.handler.click_search_btn()#点击搜索按钮业务
    
    def task_search_and_get_results(self, keyword):#搜索并获取结果业务
        self.task_search(keyword)
        return {
            'count': self.handler.get_product_items_count(),
            'names': self.handler.get_product_names(),
            'prices': self.handler.get_product_prices()
        }
    
    def task_view_product_detail(self, keyword, index=0):#搜索并查看商品详情业务
        self.task_search(keyword)
        self.handler.click_product_by_index(index)

def add_locator_methods():
    # 动态添加定位方法到UserCenter_Page类
    locators = {
        'find_username': 'username',
        'find_nav_menu': 'nav_menu',
        'find_order_manage': 'order_manage',
        'find_address_manage': 'address_manage',
        'find_favorite_manage': 'favorite_manage',
        'find_user_profile': 'user_profile',
        'find_order_items': 'order_items',
        'find_order_status': 'order_status',
        'find_address_list': 'address_list',
        'find_add_address_btn': 'add_address_btn',
        'find_favorite_items': 'favorite_items'
    }
    
    for method_name, locator_name in locators.items():
        def make_method(locator_name):
            def method(self):
                locator = getattr(self, locator_name)
                if isinstance(locator, tuple):
                    if locator_name.endswith('s'):  # 如果是复数形式，使用find_elements
                        return self.driver.find_elements(locator[0], locator[1])
                    else:
                        return self.driver.find_element(locator[0], locator[1])
                return method
        
        setattr(UserCenter_Page, method_name, make_method(locator_name))

# 调用函数添加定位方法

5. 测试用例设计

5.1 设计原则

  • 独立性：每个测试用例应该独立运行，不依赖其他测试用例的结果
  • 可重复性：测试用例应该可以重复运行，每次运行的结果应该一致
  • 全面性：测试用例应该覆盖各种正常和异常场景
  • 可读性：测试用例应该易于理解和维护

5.2 测试用例结构

每个测试用例类都包含以下方法：

  • setup_class()：测试类初始化方法，在所有测试方法执行前运行
  • teardown_class()：测试类清理方法，在所有测试方法执行后运行
  • setup_method()：测试方法初始化方法，在每个测试方法执行前运行
  • teardown_method()：测试方法清理方法，在每个测试方法执行后运行
  • 测试方法：以test_开头的方法，实现具体的测试逻辑

6. 运行方式

6.1 运行单个测试文件

python -m pytest -s Shop_UI/login.py

6.2 运行单个测试用例

python -m pytest -s Shop_UI/login.py::Test_Login::test_wrong_password

6.3 运行所有测试用例

python -m pytest -s Shop_UI/
