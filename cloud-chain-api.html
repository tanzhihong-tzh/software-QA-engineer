4. 自动化脚本示例

用户登录接口测试脚本

import pytest import requests from utils.logger import logger from utils.assertions import assert_json class TestUserLogin: """用户登录接口测试""" def setup_class(self): self.url = "https://api.cloudchain.com/api/v1/users/login" self.headers = {"Content-Type": "application/json"} def test_login_success(self): """测试登录成功""" payload = { "username": "test_user", "password": "Test@1234" } response = requests.post(self.url, json=payload, headers=self.headers) logger.info(f"登录接口响应: {response.json()}") # 断言 assert response.status_code == 200 assert_json(response.json(), { "success": True, "data": { "token": "string", "user_id": "integer" }, "message": "登录成功" }) def test_login_failure_wrong_password(self): """测试密码错误登录失败""" payload = { "username": "test_user", "password": "Wrong@1234" } response = requests.post(self.url, json=payload, headers=self.headers) logger.info(f"登录接口响应: {response.json()}") # 断言 assert response.status_code == 401 assert response.json()["success"] == False assert "密码错误" in response.json()["message"]
